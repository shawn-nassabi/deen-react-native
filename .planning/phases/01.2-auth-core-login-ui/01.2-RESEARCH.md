## Summary

Phase 1.2 performs three parallel rewrites: (1) `utils/auth.ts` — gut Cognito PKCE, install Supabase thin wrappers; (2) `hooks/useAuth.tsx` — wire `onAuthStateChange` listener, preserve the public API surface; and (3) auth screens — replace the single-button login with an email+password card form and add a new signup screen.

The Supabase client singleton (`utils/supabase.ts`) with `LargeSecureStore`, `persistSession: true`, `autoRefreshToken: true`, and AppState lifecycle wiring was fully delivered in Phase 1.1. This phase only calls into that client — no new packages are needed. The type contract between `supabase.auth.onAuthStateChange`, `supabase.auth.signInWithPassword`, `supabase.auth.signUp`, and `supabase.auth.signOut` has been verified directly against the installed `@supabase/auth-js` type definitions (v2.103.0).

The single non-trivial compatibility concern is that `onAuthStateChange` fires an `INITIAL_SESSION` event on startup (not `SIGNED_IN`), and the callback must be registered before any `await` to guarantee it captures the restored session. All other consuming screens (`_layout.tsx`, `settings.tsx`, `hikmah.tsx`, `lesson/[lessonId].tsx`, `ElaborationModal.tsx`) use only `status`, `user`, `accessToken`, `signIn`, and `signOut` from `useAuth` — all of which are preserved or have backward-compat shims.

**Primary recommendation:** Register `onAuthStateChange` synchronously in `AuthProvider` (not inside an async `useEffect`) to avoid missing the `INITIAL_SESSION` event that fires immediately on mount.

## Architecture Patterns

### Pattern 1: `utils/auth.ts` Thin-Wrapper Rewrite

The entire file is replaced. The new file is short — all logic lives in the Supabase SDK.

```typescript
// Source: verified against @supabase/auth-js 2.103.0 type definitions

/**
 * Auth utility wrappers for Deen mobile app.
 * Thin wrappers around Supabase auth calls.
 * Cognito PKCE logic removed in Phase 1.2.
 */

import { supabase } from "@/utils/supabase";

// ---- Auth wrappers ----

export async function signIn(email: string, password: string): Promise<void> {
  const { error } = await supabase.auth.signInWithPassword({ email, password });
  if (error) throw error;
}

export async function signUp(email: string, password: string): Promise<void> {
  const { error } = await supabase.auth.signUp({ email, password });
  if (error) throw error;
}

export async function signOut(): Promise<void> {
  const { error } = await supabase.auth.signOut();
  if (error) throw error;
}

// ---- Token access ----

/**
 * Returns the current Supabase access token, or null if no session.
 * Replaces manual Cognito expiry logic — Supabase SDK handles refresh.
 */
export async function getValidAccessToken(): Promise<string | null> {
  const { data: { session } } = await supabase.auth.getSession();
  return session?.access_token ?? null;
}
```

**Key difference from Cognito version:** No expiry math, no PKCE flow, no `loadTokens()`, no `saveTokens()`. The SDK owns the session lifecycle.

### Pattern 2: `hooks/useAuth.tsx` Rewrite — `onAuthStateChange` Subscription

Critical implementation details verified in installed types:

- `onAuthStateChange` returns `{ data: { subscription: Subscription } }` where `Subscription.unsubscribe: () => void`
- Callback signature: `(event: AuthChangeEvent, session: Session | null) => void`
- `AuthChangeEvent` includes: `'INITIAL_SESSION' | 'SIGNED_IN' | 'SIGNED_OUT' | 'TOKEN_REFRESHED' | 'USER_UPDATED' | 'PASSWORD_RECOVERY'`
- `Session.user: User` — `User.id: string` (UUID), `User.email?: string`
- `INITIAL_SESSION` fires synchronously on mount with the persisted session (or `null` if none)

```typescript
// Source: verified against @supabase/auth-js 2.103.0 + GoTrueClient.d.ts

export function AuthProvider({ children }: { children: ReactNode }) {
  const [status, setStatus] = useState<AuthStatus>("loading");
  const [user, setUser] = useState<AuthUser | null>(null);
  const [accessToken, setAccessToken] = useState<string | null>(null);

  useEffect(() => {
    // ---- Old token cleanup (MIG-07) ----
    SecureStore.getItemAsync("deen.auth.tokens").then((val) => {
      if (val !== null) {
        SecureStore.deleteItemAsync("deen.auth.tokens").catch((e) =>
          console.warn("Failed to clear legacy auth tokens:", e)
        );
      }
    }).catch(() => {
      // ignore — key may not exist
    });

    // ---- Subscribe to auth state ----
    // Register synchronously (not inside async) so INITIAL_SESSION is not missed.
    const { data: { subscription } } = supabase.auth.onAuthStateChange(
      (event, session) => {
        if (session) {
          setAccessToken(session.access_token);
          setUser({
            id: session.user.id,
            sub: session.user.id,           // backward compat alias
            email: session.user.email,
          });
          setStatus("signedIn");
        } else {
          setAccessToken(null);
          setUser(null);
          setStatus("signedOut");
        }
      }
    );

    return () => subscription.unsubscribe();
  }, []);

  const signIn = async (email: string, password: string) => {
    await authSignIn(email, password);
    // State updated by onAuthStateChange — no manual setUser/setStatus needed
  };

  const signUp = async (email: string, password: string) => {
    await authSignUp(email, password);
  };

  const signOut = async () => {
    await authSignOut();
    // State updated by onAuthStateChange
  };

  const refresh = async () => {
    const { data: { session } } = await supabase.auth.getSession();
    if (session) {
      setAccessToken(session.access_token);
      setUser({ id: session.user.id, sub: session.user.id, email: session.user.email });
      setStatus("signedIn");
    } else {
      setAccessToken(null);
      setUser(null);
      setStatus("signedOut");
    }
  };

  // ... context value: { status, user, accessToken, signIn, signUp, signOut, refresh }
}
```

**CRITICAL:** `onAuthStateChange` callback must NOT be async. The installed type definitions explicitly warn: "Avoid using an async function inside `onAuthStateChange` as you might end up with a deadlock." The async version is marked `@deprecated`. Use synchronous state setters only inside the callback.

### Anti-Patterns to Avoid

- **Async onAuthStateChange callback:** The installed SDK explicitly warns against and deprecates this pattern. Use synchronous state setters. If async work is needed after a state change (e.g., migration), use `setTimeout(() => asyncWork(), 0)` outside the callback lock.
- **Manual token expiry tracking:** The Supabase SDK with `autoRefreshToken: true` owns token refresh. `getValidAccessToken()` calls `getSession()` which returns the current (refreshed) token — no expiry math needed.
- **Calling `supabase.auth.getSession()` inside `onAuthStateChange`:** This will deadlock. The callback runs inside an exclusive lock; calling other SDK methods that acquire the same lock causes a hang.
- **Importing `supabase` directly in screens:** Auth screens (`login.tsx`, `signup.tsx`) should call `signIn`/`signUp` through `useAuth`, not import the Supabase client directly. This preserves the separation established in Phase 1.1.
- **Keeping `Redirect` in login.tsx:** The redirect is already handled by `_layout.tsx`. A redundant `Redirect` component on a screen is a React anti-pattern (renders during a navigation cycle) and can cause flicker.

## Don't Hand-Roll

| Problem | Don't Build | Use Instead | Why |
|---------|-------------|-------------|-----|
| Token persistence across restarts | Custom SecureStore token management | `persistSession: true` + `LargeSecureStore` (Phase 1.1) | SDK handles serialization, encryption, chunking |
| Token auto-refresh | Manual expiry check + refresh call | `autoRefreshToken: true` + AppState wiring (Phase 1.1) | SDK handles 401 retry, race conditions, background refresh scheduling |
| Auth state broadcasting | Manual state polling | `onAuthStateChange` subscription | Single source of truth; fires for all events (sign in, sign out, token refresh, app restore) |
| Session restore on cold start | `loadTokens()` + manual `getValidAccessToken()` | `INITIAL_SESSION` event from `onAuthStateChange` | SDK fires this event synchronously with the persisted session on mount |
| Password validation | Custom regex/length checks | Let the server reject — surface `error.message` | Supabase returns `AuthWeakPasswordError` with `reasons[]` for weak passwords |

**Key insight:** In the Cognito implementation, ~350 of the ~400 lines in `utils/auth.ts` were plumbing that the Supabase SDK now owns internally. The new `utils/auth.ts` is ~40 lines.

## Common Pitfalls

### Pitfall 1: Missing INITIAL_SESSION — user appears signed out on cold start

**What goes wrong:** If `onAuthStateChange` is registered after an `await` (e.g., inside an async IIFE), the `INITIAL_SESSION` event fires before the listener is registered and the user is stuck on the loading screen.

**Why it happens:** The event fires synchronously during `createClient` initialization when a persisted session exists. Any async gap before `.onAuthStateChange()` is called causes the event to be missed.

**How to avoid:** Register `onAuthStateChange` synchronously at the top of `useEffect(() => { ... }, [])`. The MIG-07 cleanup (`SecureStore.deleteItemAsync`) must be fire-and-forget (`.then().catch()`), not awaited before the subscription is established.

**Warning signs:** App shows loading spinner indefinitely or flashes to signed-out state on every restart.

### Pitfall 2: `signUp` returns null session when email confirmation is enabled

**What goes wrong:** `supabase.auth.signUp()` returns `{ data: { user, session: null }, error: null }` when "Confirm email" is enabled in the Supabase dashboard. The `onAuthStateChange` callback fires with `session = null`, so `status` stays `signedOut` — the user appears stuck on the signup screen.

**Why it happens:** The REQUIREMENTS.md notes recommend disabling email confirmation for development. If it is accidentally enabled, signup succeeds at the API level but no session is issued.

**How to avoid:** Confirm the Supabase dashboard has "Confirm email" disabled (noted in STATE.md as a project-level decision). If it must be enabled, `signup.tsx` needs to detect `session === null` after `signUp()` and show "Check your email for a confirmation link" rather than waiting for a redirect.

**Warning signs:** Signup button spins indefinitely, no navigation to main tab.

### Pitfall 3: `_layout.tsx` auth redirect loop on signup screen

**What goes wrong:** User is on `/signup` (unauthenticated). The `useEffect` in `RootNavigator` checks `segments[0] === "login"` — which is false for the signup route — and immediately redirects to `/login`. The user can never reach the signup screen.

**Why it happens:** The existing redirect logic only excludes the `login` route from redirect-to-login. Adding a second auth route without updating this check breaks navigation.

**How to avoid:** Broaden the check to `segments[0] === "login" || segments[0] === "signup"` (see Architecture Patterns). This is a required change in `_layout.tsx`.

**Warning signs:** `router.push('/signup')` from login causes an immediate redirect back to login.

### Pitfall 4: `user.sub` consumers break because `sub` is undefined

**What goes wrong:** `hikmah.tsx`, `lesson/[lessonId].tsx`, and `ElaborationModal.tsx` all use `user?.email || user?.sub` as the `user_id` passed to API calls. If the new `AuthUser` type omits `sub`, these derive `userId = undefined` and skip all progress API calls.

**Why it happens:** The new `AuthUser` is built from a Supabase `User` object which has `id` (UUID) but no `sub` field in the JS object.

**How to avoid:** D-09 is explicit: set `sub = id` when constructing the `AuthUser` from the Supabase `User`. `{ id: session.user.id, sub: session.user.id, email: session.user.email }`. This is a backward-compat shim until Phase 1.4 migrates these consumers to `user.id` directly.

**Warning signs:** Hikmah progress not loading; `userId` is `undefined` in `hikmah.tsx` despite being signed in.

### Pitfall 5: `signOut` in `utils/auth.ts` called with `opts?: { global?: boolean }` — Supabase doesn't accept this shape

**What goes wrong:** The existing `settings.tsx` calls `signOut()` from `useAuth`, which the Cognito version forwarded with `opts` to `authSignOut(opts)`. The Supabase `signOut()` has its own options shape (`{ scope?: 'global' | 'local' | 'others' }`). If the old signature is blindly forwarded, TypeScript will error.

**Why it happens:** The `useAuth` hook's `signOut` signature includes `opts?: { global?: boolean }` from the Cognito era.

**How to avoid:** D-10 says to preserve `signOut(opts?: { global?: boolean })` on the hook's public API (for backward compat) but `utils/auth.ts` `signOut()` becomes a zero-arg Supabase wrapper. The hook ignores `opts.global` for now (Phase 1.4 cleanup). The thin wrapper in `utils/auth.ts` just calls `supabase.auth.signOut()` with no args.

### Error message mapping (Claude's discretion)

```typescript
function getAuthErrorMessage(error: unknown): string {
  if (!error) return "An unexpected error occurred.";
  const msg = (error as any)?.message ?? String(error);

  if (msg.includes("Invalid login credentials")) {
    return "Incorrect email or password. Please try again.";
  }
  if (msg.includes("User already registered")) {
    return "An account with this email already exists. Sign in instead.";
  }
  if (msg.includes("weak_password") || msg.includes("Password should")) {
    return "Password is too weak. Use at least 8 characters.";
  }
  if (msg.includes("Unable to validate") || msg.includes("fetch")) {
    return "Network error. Please check your connection and try again.";
  }
  // Supabase error messages are generally user-readable — surface them directly as fallback
  return msg || "Something went wrong. Please try again.";
}
```

### MIG-07: Legacy token cleanup

```typescript
// D-15: Check and delete old Cognito token key on first AuthProvider mount
// Source: expo-secure-store v15 API (already installed)
import * as SecureStore from "expo-secure-store";

const LEGACY_TOKEN_KEY = "deen.auth.tokens";

// Fire-and-forget — do NOT await before registering onAuthStateChange

  if (val !== null) {

    SecureStore.deleteItemAsync(LEGACY_TOKEN_KEY).catch((e) =>
      console.warn("Failed to clear legacy auth tokens:", e)
    );
  }
}).catch(() => {
  // Key may not exist — silent ignore
```

## State of the Art

| Old Approach (Cognito) | New Approach (Supabase) | Impact |
|------------------------|------------------------|--------|
| Manual PKCE flow via `expo-auth-session` | `signInWithPassword({ email, password })` | ~350 lines removed from `utils/auth.ts` |
| Manual token expiry tracking + refresh logic | `autoRefreshToken: true` + `getSession()` | No expiry math; SDK owns refresh |
| Manual `loadTokens()` on app start | `INITIAL_SESSION` from `onAuthStateChange` | No async init gap; event-driven |
| `signIn()` opens a WebBrowser for Cognito hosted UI | In-app `TextInput` email+password form | No browser redirect; faster UX |
| `StoredTokens` shape in SecureStore | Supabase session JSON serialized by SDK via `LargeSecureStore` | Storage format is SDK-internal |
| `user.sub` = Cognito username/email string | `user.id` = Supabase UUID | `user.sub` aliased to `user.id` for compat until Phase 1.4 |
