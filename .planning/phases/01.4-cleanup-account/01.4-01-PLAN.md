import { Linking, type ComponentProps } from 'react-native';

type Props = Omit<ComponentProps<typeof Link>, 'href'> & { href: Href & string };

export function ExternalLink({ href, ...rest }: Props) {
  return (
    <Link
      target="_blank"
      {...rest}
      href={href}
      onPress={async (event) => {
        if (process.env.EXPO_OS !== 'web') {
          // Prevent the default behavior of linking to the default browser on native.
          event.preventDefault();
          // Open the link in the system browser (D-01: Linking.openURL replaces openBrowserAsync).

        }
      }}
    />
  );
}

**Step 1 — Add deleteAccount() to utils/api.ts (D-09).**

Append the following at the end of utils/api.ts (after the closing brace of `elaborateSelectionStream`):

```typescript

// ---- Account ----

/**
 * Delete the authenticated user's account.
 * Calls DELETE /account/me with the Supabase Bearer token.
 * Returns void on 204 success; throws a descriptive Error on failure.
 */
export async function deleteAccount(): Promise<void> {
  const authHeaders = await withAuthHeaders();
  const res = await fetch(`${API_BASE_URL}/account/me`, {
    method: "DELETE",
    headers: authHeaders,
  });
  if (!res.ok) {
    const errorText = await res.text().catch(() => res.statusText);
    throw new Error(`Failed to delete account (HTTP ${res.status}): ${errorText}`);
  }
}
```

`withAuthHeaders()` is already defined in the file and adds the Supabase Bearer token. No new imports needed.
